cmake_minimum_required(VERSION 3.28)
project(bjvm)

enable_testing()

set(CMAKE_C_STANDARD 23)
set(CMAKE_CXX_STANDARD 17)
Include(FetchContent)

FetchContent_Declare(
        Catch2
        GIT_REPOSITORY https://github.com/catchorg/Catch2.git
        GIT_TAG v3.4.0
)

FetchContent_MakeAvailable(Catch2)

if (NOT EMSCRIPTEN)
FetchContent_Declare(
        zlib
        GIT_REPOSITORY https://github.com/madler/zlib.git
        GIT_TAG        v1.3.1
)

FetchContent_MakeAvailable(zlib)
endif()

# If doing emscripten
if (EMSCRIPTEN)
    set(CMAKE_C_FLAGS "-O2 -fms-extensions -Wall -Wextra -Wpedantic -sUSE_ZLIB=1 -mtail-call")
    set(CMAKE_CXX_FLAGS "-Wall -Wpedantic")
    set(EmscriptenFlags "-s EXPORTED_FUNCTIONS=\"['_malloc','_free','_main']\" -gsource-map -sSTACK_SIZE=500000 -sALLOW_MEMORY_GROWTH=1 --emit-tsd bjvm_main.d.ts -s ALLOW_TABLE_GROWTH=1 -s EXPORTED_RUNTIME_METHODS='\"ccall\",\"wasmMemory\",\"cwrap\",\"addFunction\",\"removeFunction\",\"wasmTable\",\"FS\",\"UTF8ToString\",\"getValue\",\"setValue\"'")
elseif (APPLE)
    set(CMAKE_C_FLAGS "-O2 -fms-extensions -Wall -Wextra -Wimplicit-fallthrough -g -Werror=incompatible-pointer-types -Wno-missing-field-initializers -fno-strict-aliasing")
    set(CMAKE_CXX_FLAGS "-O2 -Wall -Wextra -Wimplicit-fallthrough -g -Werror=incompatible-pointer-types -Wno-missing-field-initializers -fno-strict-aliasing") # -openmp -Xclang -fopenmp -L/opt/homebrew/opt/libomp/lib -I/opt/homebrew/opt/libomp/include -lomp -g")
    set(EmscriptenFlags "")
else ()
    set(CMAKE_C_FLAGS "-O2 -fms-extensions -Wall -Wextra -g -Werror=incompatible-pointer-types -march=native -fno-strict-aliasing -lm")
    set(CMAKE_CXX_FLAGS "-O2 -fms-extensions -Wall -g -fopenmp -march=native -Werror=incompatible-pointer-types -fno-strict-aliasing -lm")
    set(EmscriptenFlags "")
endif ()

set(CMAKE_CXX_FLAGS_RELEASE "${CMAKE_CXX_FLAGS_DEBUG} -O3")
set(CMAKE_CXX_FLAGS_DEBUG "${CMAKE_CXX_FLAGS_DEBUG} -O2")

if (FALSE)
    set(CMAKE_C_FLAGS "${CMAKE_C_FLAGS} -DNDEBUG")
    set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -DNDEBUG")
endif ()

if (TRUE AND NOT EMSCRIPTEN)
    set(CMAKE_C_FLAGS "${CMAKE_C_FLAGS} -fsanitize=address,undefined")
    set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -fsanitize=address,undefined")
endif ()

file(GLOB bjvm_SRC CONFIGURE_DEPENDS "src/*.c" "src/*.h")
file(GLOB_RECURSE bjvm_NATIVES CONFIGURE_DEPENDS "src/natives/*.c" "src/natives/*.h")

add_library(bjvm OBJECT ${bjvm_SRC} ${bjvm_NATIVES}
        vendor/stb_ds.h
        vendor/stb_ds.c
        src/natives/java/lang/reflect/Executable.c
        src/wasm_adapter.c
        src/wasm_adapter.h
        src/natives/jdk/internal/misc/CDS.c
        src/natives/java/lang/NullPointerException.c
        src/natives/java/lang/StringUTF16.c
        src/natives/jdk/internal/misc/Signal.c
        src/natives/jdk/internal/loader/BootLoader.c
        src/natives/java/lang/Module.c)
target_include_directories(bjvm PUBLIC src ${CMAKE_CURRENT_SOURCE_DIR})

if (NOT EMSCRIPTEN)
target_link_libraries(bjvm PRIVATE zlibstatic)
else()
target_link_libraries(bjvm PRIVATE z)
endif()

add_executable(bjvm_async_test src/bin/async.c)  # for now
target_link_libraries(bjvm_async_test PRIVATE bjvm)

if (EMSCRIPTEN)
    add_executable(bjvm_main src/bin/main.c)  # for now
    target_link_libraries(bjvm_main PRIVATE bjvm)
    set_target_properties(bjvm_main PROPERTIES LINK_FLAGS "${EmscriptenFlags} -sMODULARIZE -sEXPORT_ES6=1")
endif()

if (NOT APPLE)
    set_target_properties(bjvm PROPERTIES LINK_DEPENDS ${CMAKE_SOURCE_DIR}/natives_table.ld)
endif ()

add_executable(gen_natives codegen/gen_natives.c)
target_link_libraries(gen_natives PRIVATE bjvm)
set_target_properties(gen_natives PROPERTIES LINK_FLAGS "${EmscriptenFlags}")

add_executable(tests test/tests.cc test/arrays-test.cc test/tests-common.cc test/wasm-tests.cc
        test/async-tests.cc
        test/analysis-tests.cc
        test/classpath-tests.cc
        test/benches.cc)
target_link_libraries(tests PRIVATE bjvm)
target_link_libraries(tests PRIVATE Catch2::Catch2WithMain)
set_target_properties(tests PROPERTIES LINK_FLAGS "${EmscriptenFlags}")

add_test(NAME tests
        COMMAND tests --skip-benchmarks
        WORKING_DIRECTORY "${CMAKE_SOURCE_DIR}/test")
set_tests_properties(tests PROPERTIES ENVIRONMENT "ASAN_OPTIONS=detect_leaks=1,detect_container_overflow=0")
